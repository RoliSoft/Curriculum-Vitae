% create new hlink command; this is needed, since \httplink only supports http:// links.

\makeatletter
\newcommand*{\hlink}[2][]{%
	\ifthenelse{\equal{#1}{}}%
	{\href{#2}{#2}}%
	{\href{#2}{#1}}}
\renewcommand*{\httplink}[2][]{%
	\ifthenelse{\equal{#1}{}}%
	{\href{https://#2}{#2}}%
	{\href{https://#2}{#1}}}
\makeatother
